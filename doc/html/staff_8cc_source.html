105  // infered. The straight line will be the chosen model
106  std::vector<cv::Vec2f> lines;
107  cv::HoughLines(img, lines, 1, CV_PI / (180 * THETA_RES), img.cols / 2);
108

111  std::vector<cv::Vec2f>());
112  for (auto line : lines) {
113  const double theta = line[1] * RAD2DEG;
114  const int idx = (int)theta / (int)(CV_PI * RAD2DEG / N_BINS);
115  theta_count[idx].push_back(line);
116  }
117

120  for (int i = 0; i < theta_count.size(); i++) {
121  if (theta_count[i].size() > max_count) {
122  max_index = i;
123  max_count = theta_count[i].size();
124  }
125  }
126

130  for (auto line : theta_count[max_index]) {

132  avg_theta += theta;

358  cv::Mat staff_image = model.staff_image;
359 
360  // Interpolating empty columns
361  double prev_orientation = staff_image.rows,

372  break;
373  }
374  i++;

377  if ((prev_orientation == staff_image.rows) &&
378  (next_orientation != staff_image.rows)) {

381  (next_orientation == staff_image.rows)) {
382  next_orientation = prev_orientation;
383  }

385  (next_orientation - prev_orientation) / (i - current);

718 void as::staff::Realign(cv::Mat &dst, const StaffModel &model) {
719  assert(is_gray(dst));
720  assert(dst.cols >= model.gradient.size());
721 
722  if (model.straight) {
723  return;
724  }
725 
726  const bool blackOnWhite = black_on_white(dst);
727  if (blackOnWhite)
728  cv::threshold(dst, dst, BINARY_THRESH_VAL, 255, CV_THRESH_BINARY_INV);
729  else {
730  cv::threshold(dst, dst, 255 - BINARY_THRESH_VAL, 255, CV_THRESH_BINARY);
731  }
732  const double rotation = RAD2DEG * (model.rot - CV_PI / 2);
733  rotate_image(dst, rotation);
734 
735  cv::Mat img = dst.clone();
736 
737  double start = 0;
738  for (int x = 0; x < model.gradient.size(); x++, start += model.gradient[x]) {
739  const int rounded_start = round(start);
740  const int col = x + model.start_col;
741  for (int y = 0; y < img.rows; y++) {
742  const int offset_y = y + rounded_start;
743  if (offset_y >= img.rows || offset_y < 0) {
744  img.at<char>(y, col) = 0;
745  } else {
746  img.at<char>(y, col) = dst.at<char>(offset_y, col);
747  }
748  }
749  }
750  dst = img;
751  if (blackOnWhite)
752  cv::threshold(dst, dst, BINARY_THRESH_VAL, 255, CV_THRESH_BINARY_INV);
753  raw_rotate(dst, -rotation);
754 }
